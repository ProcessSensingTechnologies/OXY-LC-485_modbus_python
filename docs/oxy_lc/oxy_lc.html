501    def calibrate(self, calibration_value: float | None) -> bool:
502        """
503        Calibrate the sensor to the given percent value
504
505        :param calibration_precent: O2 percent for calibration, defaults to 20.7
506        :type calibration_precent: float, optional
507        :return: boolean indication of success
508        :rtype: bool
509        """
510        if self.status != self.StatusValues.OPERATING:
511            print("Sensor must be in operation to calibrate")
512            return False
513
514        if calibration_value:
515            try:
516                self.calibration_value(calibration_value)
517            except ValueRangeError as e:
518                print(e)
519                return False
520
521        self.calibration_control(self.CalibrationControl.ACTIVATE)
522
523        count = 0
524        timeout = 5
525        while (self.calibration_status != self.CalibrationStatus.COMPLETED) and (
526            count != timeout
527        ):
528            sleep(1)
529            count += 1
530
531        self.write_register(
532            HoldingRegister.CALIBRATION_CONTROL, self.CalibrationControl.RESET
533        )
534
535        if count == timeout:
536            print("calibration timeout")
537            return False
538        else:
539            return True

547    def set_and_apply_heater_voltage(self, set_voltage: HeaterOptions | None) -> None:
548        """
549        Set the current heater voltage for the sensor
550
551        :param state: State option from Enum HeaterOptions
552        :type state: HeaterOptions
553        """
554        if set_voltage:
555            self.heater_voltage = set_voltage
556
557        # MinimalModbus will throw a NoResonseError when this register is written.
558        # It must be passed or it will crash the program
559        try:
560            self.write_register(
561                HoldingRegister.HEATER_VOLTAGE_SAVE, self.SaveAndApply.APPLY
562            )
563        except minimalmodbus.NoResponseError:
564            sleep(2)
565
566        self.sensor_state = self.SensorState.ON

569    def display_warnings(self) -> dict[str:bool]:
570        """
571        Reads and decodes the warnings and returns a dictionary with each warning/error with a corresponding boolean showing error state
572
573        :return: Warning states for each of the representative bits (True is error)
574        :rtype: dict[str: bool]
575        """
576        warning_states = {
577            "Pump Error": False,
578            "Heater Voltage Error": False,
579            "Asymmetry Warning": False,
580            "O2 Low Warning": False,
581            "Pressure Sensor Warning": False,
582            "Pressure Sensor Error": False,
583        }
584        warnings_bits = self.warnings
585
586        for count, state in enumerate(warning_states):
587            if warnings_bits[::-1][count] == '1':
588                warning_states[state] = True
589
590        return warning_states

503        """
504        Calibrate the sensor to the given percent value
505
506        :param calibration_precent: O2 percent for calibration, defaults to 20.7
507        :type calibration_precent: float, optional
508        :return: boolean indication of success
509        :rtype: bool
510        """
511        if self.status != self.StatusValues.OPERATING:
512            print("Sensor must be in operation to calibrate")
513            return False
514
515        if calibration_value:
516            try:
517                self.calibration_value(calibration_value)
518            except ValueRangeError as e:
519                print(e)
520                return False
521
522        self.calibration_control(self.CalibrationControl.ACTIVATE)
523
524        count = 0
525        timeout = 5
526        while (self.calibration_status != self.CalibrationStatus.COMPLETED) and (
527            count != timeout
528        ):
529            sleep(1)
530            count += 1
531
532        self.write_register(
533            HoldingRegister.CALIBRATION_CONTROL, self.CalibrationControl.RESET
534        )

540            return True

570    def display_warnings(self) -> dict[str:bool]:
571        """
572        Reads and decodes the warnings and returns a dictionary with each warning/error with a corresponding boolean showing error state
573
574        :return: Warning states for each of the representative bits (True is error)
575        :rtype: dict[str: bool]
576        """
577        warning_states = {
578            "Pump Error": False,
579            "Heater Voltage Error": False,
580            "Asymmetry Warning": False,
581            "O2 Low Warning": False,
582            "Pressure Sensor Warning": False,
583            "Pressure Sensor Error": False,
584        }
585        warnings_bits = self.warnings
586
587        for count, state in enumerate(warning_states):
588            if warnings_bits[::-1][count] == '1':
589                warning_states[state] = True
590
591        return warning_states
